function createTodoApp() {
    // Tab 키만 누르면...
    const todos = [];
    
    function addTodo(text) {
        todos.push({
            id: Date.now(),
            text,
            completed: false
        });
    }
    
    function toggleTodo(id) {
        const todo = todos.find(t => t.id === id);
        if (todo) {
            todo.completed = !todo.completed;
        }
    }
}
